# mini Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a two-file minimal rundown experiment — a Python stdlib HTTP server and a single HTML file — that renders a markdown file, lets you run shell snippets in-place, and supports inline editing.

**Architecture:** `server.py` exposes five routes (serve HTML, get content, run snippet, save file, SSE events) and contains all server logic including regex-based output block manipulation. `index.html` is self-contained: fetches raw markdown, renders with marked.js, post-processes code blocks to inject Run buttons and click-to-edit behaviour, and listens for SSE reload events.

**Tech Stack:** Python 3 stdlib only (`http.server`, `subprocess`, `threading`, `re`); marked.js@14 from unpkg CDN; vanilla JS.

def update_output_block(content: str, n: int, output_text: str) -> str:
    """Insert or replace the output fence after the Nth shell block (0-indexed)."""
    blocks = find_shell_blocks(content)
    if n >= len(blocks):
        raise ValueError(f'No shell block at index {n} (found {len(blocks)})')
    block = blocks[n]
    after = content[block.end():]
    new_output = f'\n\n```output\n{output_text}\n```'
    m = _OUTPUT_AFTER_RE.match(after)
    if m:
        return content[:block.end()] + new_output + after[m.end():]
    else:
        return content[:block.end()] + new_output + after
```

```python
# ── Global state ──────────────────────────────────────────────────────────────
_md_path: Path = None          # set in __main__
_suppress = False              # True while server is writing the file
_sse_clients: list = []        # list of wfile objects waiting for events
_last_mtime: float = 0.0
_lock = threading.Lock()       # guards _suppress and _sse_clients mutations


# ── Request handler ───────────────────────────────────────────────────────────
class Handler(BaseHTTPRequestHandler):
    def log_message(self, fmt, *args):
        pass  # suppress default access log

    # ── helpers ───────────────────────────────────────────────────────────────
    def _send(self, code: int, ctype: str, body: bytes):
        self.send_response(code)
        self.send_header('Content-Type', ctype)
        self.send_header('Content-Length', str(len(body)))
        self.end_headers()
        self.wfile.write(body)

    def _read_body(self) -> bytes:
        length = int(self.headers.get('Content-Length', 0))
        return self.rfile.read(length) if length else b''

    def _parse_form(self) -> dict:
        from urllib.parse import parse_qs
        raw = self._read_body()
        parsed = parse_qs(raw.decode())
        return {k: v[0] for k, v in parsed.items()}

    # ── GET / ─────────────────────────────────────────────────────────────────
    def _get_root(self):
        html_path = Path(__file__).parent / 'index.html'
        body = html_path.read_bytes()
        self._send(200, 'text/html; charset=utf-8', body)

    # ── GET /content ──────────────────────────────────────────────────────────
    def _get_content(self):
        body = _md_path.read_text().encode()
        self._send(200, 'text/plain; charset=utf-8', body)

    # ── GET /events ───────────────────────────────────────────────────────────
    def _get_events(self):
        self.send_response(200)
        self.send_header('Content-Type', 'text/event-stream')
        self.send_header('Cache-Control', 'no-cache')
        self.end_headers()
        with _lock:
            _sse_clients.append(self.wfile)
        try:
            # Keep the connection alive with periodic pings; detect disconnect
            # on write failure (BrokenPipeError / OSError).
            while True:
                time.sleep(1)
                self.wfile.write(b': ping\n\n')
                self.wfile.flush()
        except (BrokenPipeError, OSError):
            pass
        finally:
            with _lock:
                if self.wfile in _sse_clients:
                    _sse_clients.remove(self.wfile)

    # ── POST /run ─────────────────────────────────────────────────────────────
    def _post_run(self):
        form = self._parse_form()
        try:
            n = int(form['index'])
        except (KeyError, ValueError):
            self._send(400, 'text/plain', b'Missing or invalid index')
            return

        content = _md_path.read_text()
        blocks = find_shell_blocks(content)
        if n >= len(blocks):
            self._send(400, 'text/plain', f'No block at index {n}'.encode())
            return

        code = blocks[n].group()
        # Strip the fence lines, keep only the code body
        code_body = '\n'.join(code.splitlines()[1:-1])

        try:
            proc = subprocess.run(
                ['sh', '-c', code_body],
                capture_output=True, text=True, timeout=TIMEOUT
            )
            output = (proc.stdout + proc.stderr).rstrip()
        except subprocess.TimeoutExpired:
            output = f'Error: timed out after {TIMEOUT}s'

        updated = update_output_block(content, n, output)
        with _lock:
            global _suppress
            _suppress = True
        try:
            _md_path.write_text(updated)
        finally:
            with _lock:
                _suppress = False

        self._send(200, 'text/plain; charset=utf-8', output.encode())

    # ── POST /save ────────────────────────────────────────────────────────────
    def _post_save(self):
        body = self._read_body().decode()
        with _lock:
            global _suppress
            _suppress = True
        try:
            _md_path.write_text(body)
        finally:
            with _lock:
                _suppress = False
        self._send(200, 'text/plain', b'ok')

    # ── dispatch ──────────────────────────────────────────────────────────────
    def do_GET(self):
        if self.path == '/':
            self._get_root()
        elif self.path == '/content':
            self._get_content()
        elif self.path == '/events':
            self._get_events()
        else:
            self._send(404, 'text/plain', b'Not found')

    def do_POST(self):
        if self.path == '/run':
            self._post_run()
        elif self.path == '/save':
            self._post_save()
        else:
            self._send(404, 'text/plain', b'Not found')
```

# ── File watcher ──────────────────────────────────────────────────────────────
def _watch():
    global _last_mtime
    _last_mtime = _md_path.stat().st_mtime
    while True:
        time.sleep(1)
        try:
            mtime = _md_path.stat().st_mtime
        except FileNotFoundError:
            continue
        if mtime != _last_mtime:
            _last_mtime = mtime
            with _lock:
                if _suppress:
                    continue
                clients = list(_sse_clients)
            dead = []
            for wfile in clients:
                try:
                    wfile.write(b'data: change\n\n')
                    wfile.flush()
                except (BrokenPipeError, OSError):
                    dead.append(wfile)
            if dead:
                with _lock:
                    for d in dead:
                        if d in _sse_clients:
                            _sse_clients.remove(d)


# ── Entry point ───────────────────────────────────────────────────────────────
if __name__ == '__main__':
    from http.server import ThreadingHTTPServer
    if len(sys.argv) < 2:
        print('Usage: python3 server.py <file.md>', file=sys.stderr)
        sys.exit(1)
    _md_path = Path(sys.argv[1])
    if not _md_path.exists():
        _md_path.write_text(f'# {_md_path.stem}\n')
    print(f'Serving {_md_path} at http://localhost:8000')
    threading.Thread(target=_watch, daemon=True).start()
    ThreadingHTTPServer(('', 8000), Handler).serve_forever()
```

<head>

<body>
<div id="content"></div>
<script>
  var rawSource = '';
  var snippetIndex = 0; // incremented during post-process, reset each render

  // ── fetch + render ──────────────────────────────────────────────────────────
  function load() {
    return fetch('/content')
      .then(function(r) { return r.text(); })
      .then(function(src) {
        rawSource = src;
        render();
      });
  }

  function render() {
    snippetIndex = 0;
    var html = marked.parse(rawSource);
    var div = document.getElementById('content');
    div.innerHTML = html;
    postProcess(div);
  }

  // ── post-process: wrap cells, inject Run buttons ────────────────────────────
  function postProcess(root) {
    // Wrap every direct child of #content in a data-cell div.
    // Output fences (language-output) are removed from the DOM — their content
    // is already shown in the preceding snippet's output div.
    // Both this walk and splitRawBlocks skip output fences, keeping indices aligned.
    var children = Array.from(root.children);
    var cellIndex = 0;
    _rawBlocks = null; // reset cache for this render
    var rawBlocks = splitRawBlocks(rawSource); // excludes output fences

    children.forEach(function(el) {
      var code = el.querySelector('code');
      var lang = code ? code.className.replace('language-', '') : '';

      // Remove output fence elements — content shown via extractExistingOutput
      if (lang === 'output') {
        el.remove();
        return;
      }

      var wrapper = document.createElement('div');
      wrapper.dataset.cell = cellIndex;
      wrapper.dataset.raw = rawBlocks[cellIndex] !== undefined ? rawBlocks[cellIndex] : '';
      cellIndex++;

      el.parentNode.insertBefore(wrapper, el);
      wrapper.appendChild(el);

      if (['sh', 'bash', 'shell'].indexOf(lang) !== -1) {
        var idx = snippetIndex++;
        wrapper.classList.add('snippet');
        var btn = document.createElement('button');
        btn.textContent = 'Run';
        btn.dataset.idx = idx;
        btn.addEventListener('click', function(e) {
          e.stopPropagation();
          runSnippet(parseInt(btn.dataset.idx), wrapper);
        });
        var out = document.createElement('div');
        out.className = 'output';
        out.dataset.outputFor = idx;
        var outputText = extractExistingOutput(idx);
        if (outputText !== null) out.textContent = outputText;
        wrapper.appendChild(btn);
        wrapper.appendChild(out);
      }
    });

    // Click-to-edit on non-button clicks
    root.addEventListener('click', function(e) {
      if (e.target.tagName === 'BUTTON') return;
      if (root.querySelector('[data-cell].editing')) return;
      var cell = e.target.closest('[data-cell]');
      if (cell) startEditing(cell);
    });
  }

  // ── raw source helpers ──────────────────────────────────────────────────────
  // Split rawSource into top-level blocks, skipping output fences.
  // "Block" = consecutive non-blank lines; separated by one or more blank lines.
  // Fenced code blocks are kept intact (fence toggling tracks open/close).
  // Output fences (```output...```) are excluded so the index matches postProcess.
  var _rawBlocks = null;

  function splitRawBlocks(src) {
    var blocks = [];
    var current = '';
    var inFence = false;
    var fenceLang = '';
    var lines = src.split('\n');
    for (var i = 0; i < lines.length; i++) {
      var line = lines[i];
      var fenceOpen = !inFence && /^```(\w*)/.exec(line);
      var fenceClose = inFence && /^```\s*$/.test(line);
      if (fenceOpen) {
        inFence = true;
        fenceLang = fenceOpen[1];
      }
      if (!inFence && line.trim() === '' && current.trim() !== '') {
        // End of a block — only push if not an output fence
        if (!/^```output\n/.test(current.trim())) {
          blocks.push(current);
        }
        current = '';
      } else {
        current += (current === '' ? '' : '\n') + line;
      }
      if (fenceClose) {
        inFence = false;
        // If this closed an output fence, discard it
        if (fenceLang === 'output') {
          current = '';
        }
        fenceLang = '';
      }
    }
    if (current.trim() !== '' && !/^```output\n/.test(current.trim())) {
      blocks.push(current);
    }
    return blocks;
  }

  function extractExistingOutput(snippetIdx) {
    // Find the Nth shell fence in rawSource, check for an output fence after it
    var shellRe = /```(?:sh|bash|shell)\n[\s\S]*?```/g;
    var matches = [];
    var m;
    while ((m = shellRe.exec(rawSource)) !== null) matches.push(m);
    if (snippetIdx >= matches.length) return null;
    var after = rawSource.slice(matches[snippetIdx].index + matches[snippetIdx][0].length);
    var outM = after.match(/^\n\n?```output\n([\s\S]*?)```/);
    return outM ? outM[1].replace(/\n$/, '') : null;
  }

  // ── run ─────────────────────────────────────────────────────────────────────
  function runSnippet(idx, wrapper) {
    var out = wrapper.querySelector('[data-output-for="' + idx + '"]');
    if (out) out.textContent = 'running…';
    var body = new URLSearchParams({ index: idx });
    fetch('/run', { method: 'POST', body: body })
      .then(function(r) { return r.text(); })
      .then(function(text) {
        if (out) out.textContent = text;
      });
  }

  // ── edit ────────────────────────────────────────────────────────────────────
  function startEditing(cell) {
    var raw = cell.dataset.raw || '';
    var origHTML = cell.innerHTML;
    cell.classList.add('editing');
    cell.innerHTML = '<textarea></textarea><div class="edit-hint">⌘↵ save · Esc cancel</div>';
    var ta = cell.querySelector('textarea');
    ta.value = raw;
    ta.style.height = 'auto';
    ta.style.height = ta.scrollHeight + 'px';
    ta.focus();

    ta.addEventListener('keydown', function(e) {
      if (e.key === 'Escape') {
        cell.classList.remove('editing');
        cell.innerHTML = origHTML;
      }
      if ((e.metaKey || e.ctrlKey) && e.key === 'Enter') {
        e.preventDefault();
        saveEdit(cell, raw, ta.value);
      }
    });
  }

  function saveEdit(cell, oldRaw, newRaw) {
    // Replace first occurrence of oldRaw literally (indexOf avoids $-pattern hazard of .replace)
    var pos = rawSource.indexOf(oldRaw);
    var newSource = pos === -1 ? rawSource
      : rawSource.slice(0, pos) + newRaw + rawSource.slice(pos + oldRaw.length);
    fetch('/save', { method: 'POST', body: newSource, headers: { 'Content-Type': 'text/plain' } })
      .then(function(r) {
        if (!r.ok) {
          alert('Save failed');
          return;
        }
        return load();
      });
  }

  // ── SSE ─────────────────────────────────────────────────────────────────────
  var es = new EventSource('/events');
  es.addEventListener('message', function(e) {
    if (e.data === 'change') load();
  });

  // ── init ────────────────────────────────────────────────────────────────────
  load();
</script>
</body>
</html>
```
